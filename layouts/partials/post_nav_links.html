{{/* Series-aware prev/next navigation. Falls back to default order if no series. */}} {{- $series := .Params.series -}} {{- if and $series (ge (len $series) 1) -}} {{- $seriesPages := where site.RegularPages "Params.series" "intersect" $series -}} {{- $seriesPages = sort $seriesPages "Params.series_order" "asc" -}} {{- $cur := . -}} {{- $idx := -1 -}} {{- range $i, $p := $seriesPages -}} {{- if eq $p.Permalink $cur.Permalink -}} {{- $idx = $i -}} {{- end -}} {{- end -}} {{- if ge $idx 0 -}}
{{- if ge $idx 1 -}} {{- $prev := index $seriesPages (sub $idx 1) -}} « {{ i18n "prev_page" }}
{{- $prev.Title -}} {{- end -}} {{- if lt (add $idx 1) (len $seriesPages) -}} {{- $next := index $seriesPages (add $idx 1) -}} {{ i18n "next_page" }} »
{{- $next.Title -}} {{- end -}}
{{- end -}} {{- else -}} {{- /* Fallback to theme default behavior */ -}} {{- $pages := where site.RegularPages "Type" "in" site.Params.mainSections }} {{- if and (gt (len $pages) 1) (in $pages . ) }}
{{- with $pages.Next . }} « {{ i18n "prev_page" }}
{{- .Name -}} {{- end }} {{- with $pages.Prev . }} {{ i18n "next_page" }} »
{{- .Name -}} {{- end }}
{{- end }} {{- end -}}
